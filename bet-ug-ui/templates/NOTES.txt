1. Get the application URL by running these commands:

{{- if contains "NodePort" .Values.service.type }}
export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "bet-ug-ui.fullname" . }})
export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
echo http://$NODE_IP:$NODE_PORT

{{- else if contains "LoadBalancer" .Values.service.type }}
NOTE: It may take a few minutes for the LoadBalancer IP to be available.
You can monitor it with:
kubectl get svc -n {{ .Release.Namespace }} -w {{ include "bet-ug-ui.fullname" . }}

export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "bet-ug-ui.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
echo http://$SERVICE_IP:{{ .Values.service.port }}

{{- else if contains "ClusterIP" .Values.service.type }}
NOTE: The service is exposed internally. Access it through your configured Gateway.

export SERVICE_NAME={{ include "bet-ug-ui.fullname" . }}
export NAMESPACE={{ .Release.Namespace }}

echo "Service reachable inside cluster at:"
echo "http://$SERVICE_NAME.$NAMESPACE.svc.cluster.local:{{ .Values.service.port }}"

echo ""
echo "Or use port-forward for local testing:"
export POD_NAME=$(kubectl get pods --namespace $NAMESPACE -l "app.kubernetes.io/name={{ include "bet-ug-ui.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
kubectl --namespace $NAMESPACE port-forward $POD_NAME 8080:{{ .Values.service.port }}

echo "Visit http://127.0.0.1:8080"
{{- end }}
